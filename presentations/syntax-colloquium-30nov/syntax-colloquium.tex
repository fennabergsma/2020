\begin{frame}{Introduction --- (headless) relative clauses}

\exg. Der Lehrer mag den Schüler, der nach draußen guckt.\\
the.\tsc{nom} teacher likes the.\tsc{acc} student \tsc{rel}.\tsc{sg}.\tsc{m}.\tsc{nom} to outside looks\\
`The teacher likes the pupil that is looking outside.'\label{ex:german-relative-1}

\pause

\exg. Der Lehrer mag den Schüler, den er beim Versteckspiel sucht.\\
the.\tsc{nom} teacher likes the.\tsc{acc} student \tsc{rel}.\tsc{sg}.\tsc{m}.\tsc{acc} he {at the} {hide-and-seek game} searches\\
`The teacher likes the pupil that he is searching for playing hide-and-seek.'\label{ex:german-relative-2}

\pause

\exg. Der Lehrer mag, wen er beim Versteckspiel sucht.\\
the.\tsc{nom} teacher likes \tsc{rel}.\tsc{sg}.\tsc{m}.\tsc{acc} he {at the} {hide-and-seek game} searches\\
`The teacher likes the pupil that he is searching for playing hide-and-seek.'\label{ex:german-relative-2}

\end{frame}

\begin{frame}{Introduction --- case competition}

\exg. gaarma þan -ei arma\\
 pity.\tsc{pres}.1\tsc{sg}\scsub{[acc]} \tsc{rel}.\tsc{sg}.\tsc{m}.\tsc{acc} -\tsc{comp} pity.\tsc{pres}.1\tsc{sg}\scsub{[acc]}\\
 `I pity him whom I pity' \flushfill{Gothic, \tsc{rom} 9:15, adapted from \pgcitealt{harbert1978}{339}}\label{ex:gothic-acc-acc}

\pause

\exg. \tbf{þan} \tbf{-ei} \tbf{frijos} siuks ist\\
 \tsc{rel}.\tsc{sg}.\tsc{m}.\tsc{acc} -\tsc{comp} love.\tsc{pres}.2\tsc{sg}.\scsub{[acc]} sick be.\tsc{pres}.3\tsc{sg}\scsub{[nom]}\\
 `the one whom you love is sick' \flushfill{Gothic, \tsc{john} 11:3, adapted from \pgcitealt{harbert1978}{342}}\label{ex:gothic-acc-nom}

\end{frame}
